  <action>
    Create new file `hooks/use-idle-timer.ts`:

    ```typescript
    import { useCallback, useEffect, useRef, useState } from "react";

    export function useIdleTimer(
      timeoutMs: number,
      enabled: boolean,
    ): boolean {
      const [isIdle, setIsIdle] = useState(false);
      const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

      const resetTimer = useCallback(() => {
        if (!enabled) return;
        if (timerRef.current) {
          clearTimeout(timerRef.current);
        }
        setIsIdle(false);
        timerRef.current = setTimeout(() => setIsIdle(true), timeoutMs);
      }, [enabled, timeoutMs]);

      useEffect(() => {
        if (!enabled) {
          setIsIdle(false);
          return undefined;
        }

        resetTimer();

        const events = ["mousemove", "click", "keydown"] as const;
        events.forEach((event) => {
          window.addEventListener(event, resetTimer, { passive: true });
        });

        return () => {
          if (timerRef.current) {
            clearTimeout(timerRef.current);
          }
          events.forEach((event) => {
            window.removeEventListener(event, resetTimer);
          });
        };
      }, [enabled, resetTimer]);

      return isIdle;
    }
    ```

    Key decisions documented inline (why, not what):
    - `setIsIdle(false)` on every event is safe because React dedupes same-value state updates — only the `true → false` transition triggers a reconciler pass
    - `useRef` for the timer avoids re-render on `clearTimeout`/`setTimeout`
    - `{ passive: true }` prevents scroll-jank (not critical on TV but good practice)
    - Three events cover desktop (mousemove + click), TV remotes (click/keydown), and keyboard users (keydown)

    Project conventions:
    - Double quotes, 2-space indent, semicolons, trailing commas
    - Strict TS
    - No `any`
    - Import from `react` (no star imports)
